var SCRIPT_URL = localStorage.getItem('ct_script_url') || 'https://script.google.com/macros/s/AKfycbyJNYvF7c4XwfEezPJ3tXZtmz6HJSTcCroczZInjBsEgeoI-qX3jinn4SzHyhp8-FY4hg/exec';

// normalize dates on load
try{ appts = appts.map(function(a){ if(a.date && !/^\d{4}-\d{2}-\d{2}$/.test(String(a.date))){ var d=new Date(a.date); if(!isNaN(d)&&d.getFullYear()>1900) a.date=d.toISOString().slice(0,10); } return a; }); }catch(e){}
var editClientId = null, editApptId = null, wizClientId = null, afterNewClient = false;

// ── HELPERS ─────────────────────────────────────────────
function uid(){ return Date.now().toString(36) + Math.random().toString(36).slice(2); }
function today(){ return new Date().toISOString().slice(0,10); }
function initials(n){ return n.trim().split(/\s+/).map(function(w){return w[0];}).join('').toUpperCase().slice(0,2); }
function clientById(id){ return clients.find(function(x){return x.id===id;}); }
function clientName(id){ var c=clientById(id); return c?c.name:'?'; }
function saveLocal(){
  localStorage.setItem('ct_clients', JSON.stringify(clients));
  localStorage.setItem('ct_appts',   JSON.stringify(appts));
}

// ── SYNC DOT ────────────────────────────────────────────
function setSyncDot(state){
  var d = document.getElementById('sync-dot');
  d.className = 'sync-dot ' + state;
}

// ── TOAST ───────────────────────────────────────────────
function toast(msg, dur){
  var t = document.getElementById('toast');
  t.textContent = msg; t.classList.add('show');
  setTimeout(function(){ t.classList.remove('show'); }, dur||2200);
}

// ── API CALLS ───────────────────────────────────────────
function api(params){
  setSyncDot('syncing');
  // Add timestamp to prevent caching
  var url = SCRIPT_URL + '?t=' + Date.now();
  return fetch(url, {
    method: 'POST',
    headers: {'Content-Type': 'text/plain'},
    body: JSON.stringify(params),
    cache: 'no-store'
  })
  .then(function(r){ return r.json(); })
  .then(function(data){
    setSyncDot('ok');
    setTimeout(function(){ setSyncDot(''); }, 2000);
    return data;
  })
  .catch(function(err){
    setSyncDot('err');
    throw err;
  });
}

function normalizeAppts(arr){
  return (arr||[]).map(function(a){
    if(!a.date) return a;
    var ds = String(a.date).trim();
    // Already YYYY-MM-DD
    if(/^\d{4}-\d{2}-\d{2}$/.test(ds)){ return a; }
    // Try parsing as date string
    var d = new Date(ds);
    if(!isNaN(d) && d.getFullYear() > 1900){
      a.date = d.toISOString().slice(0,10);
    }
    return a;
  });
}

function loadAll(){
  if(!SCRIPT_URL){ showSetup(); return; }
  showLoading();
  // Always fetch fresh from Google Sheets
  api({action:'getAll'}).then(function(data){
    if(data.clients) clients = data.clients;
    if(data.appts)   appts   = normalizeAppts(data.appts);
    saveLocal(); // update local cache
    hideLoading();
    showTab('home');
  }).catch(function(){
    // Offline fallback — use local cache
    hideLoading();
    toast('⚠️ Offline — lokale data geladen');
    showTab('home');
  });
}

// ── CSV IMPORT ───────────────────────────────────────────
function importCSV(e){
  var file = e.target.files[0]; if(!file) return;
  var reader = new FileReader();
  reader.onload = function(ev){
    var text = ev.target.result;
    var count=0, skip=0, batch=[];

    // Detect HTML/XLS format (CleanPlanner exports as HTML table with .xls extension)
    if(text.indexOf('<table') > -1 || text.indexOf('<TABLE') > -1 || text.indexOf('<th') > -1){
      // Parse as HTML table
      var parser = new DOMParser();
      var doc = parser.parseFromString(text, 'text/html');
      var rows = doc.querySelectorAll('tr');
      if(rows.length < 2){ alert('Geen klanten gevonden in het bestand.'); return; }

      // Get headers from first row
      var headerCells = rows[0].querySelectorAll('th,td');
      var headers = [];
      headerCells.forEach(function(c){ headers.push(c.textContent.trim().toLowerCase()); });

      function colIdx(names){
        for(var ni=0;ni<names.length;ni++){
          var idx = headers.findIndex(function(h){return h.includes(names[ni]);});
          if(idx >= 0) return idx;
        }
        return -1;
      }
      function getCell(cells, names){
        var i = colIdx(names);
        if(i<0 || !cells[i]) return '';
        return cells[i].textContent.trim();
      }

      for(var i=1;i<rows.length;i++){
        var cells = rows[i].querySelectorAll('td');
        if(cells.length < 2) continue;
        var name = getCell(cells, ['bedrijf','naam','name','commerci']);
        if(!name){ skip++; continue; }
        var exists = clients.find(function(c){return c.name.toLowerCase()===name.toLowerCase();});
        if(!exists){
          var pc  = getCell(cells, ['pc','postcode']);
          var gem = getCell(cells, ['gemeente']);
          var city = pc&&gem ? pc+' '+gem : (pc||gem);
          var phone = getCell(cells, ['tel','telefoon']);
          var gsm   = getCell(cells, ['gsm','mob']);
          var newClient = {
            id: uid(), name: name,
            addr:  getCell(cells, ['adres','address','straat']),
            city:  city,
            phone: gsm || phone,
            email: getCell(cells, ['email','e-mail','mail']),
            defaultPrice: getCell(cells, ['standaardprijs','prijs','price']),
            notes: ''
          };
          clients.push(newClient);
          batch.push(newClient);
          count++;
        } else { skip++; }
      }
    } else {
      // Parse as CSV
      var sep = text.indexOf(';') > -1 ? ';' : ',';
      var lines = text.split('\n').map(function(l){return l.trim();}).filter(Boolean);
      if(lines.length < 2){ alert('Bestand lijkt leeg.'); return; }
      var header = lines[0].split(sep).map(function(h){return h.trim().toLowerCase().replace(/["\r]/g,'');});
      function colIdxCSV(names){
        for(var ni=0;ni<names.length;ni++){
          var idx = header.findIndex(function(h){return h.includes(names[ni]);});
          if(idx >= 0) return idx;
        }
        return -1;
      }
      function getColCSV(cols, names){
        var i = colIdxCSV(names);
        if(i<0 || !cols[i]) return '';
        return cols[i].trim().replace(/^["']|["']$/g,'').trim();
      }
      for(var i=1;i<lines.length;i++){
        var cols = lines[i].split(sep);
        var name = getColCSV(cols,['bedrijf','bedrijfsnaam','naam','name','commerci']);
        if(!name){ skip++; continue; }
        var exists = clients.find(function(c){return c.name.toLowerCase()===name.toLowerCase();});
        if(!exists){
          var rawpc = getColCSV(cols,['pc','postcode']);
          var gem   = getColCSV(cols,['gemeente']);
          var city  = rawpc&&gem ? rawpc+' '+gem : (rawpc||gem);
          var newClient = {
            id: uid(), name: name,
            addr:  getColCSV(cols,['adres','address','straat']),
            city:  city,
            phone: getColCSV(cols,['gsm','tel','phone','telefoon','mob']),
            email: getColCSV(cols,['email','e-mail','mail']),
            defaultPrice: getColCSV(cols,['standaardprijs','prijs','price']),
            notes: ''
          };
          clients.push(newClient);
          batch.push(newClient);
          count++;
        } else { skip++; }
      }
    }

    saveLocal();
    if(batch.length){
      api({action:'importClients', clients: batch})
        .then(function(){ toast('✓ '+count+' klanten geïmporteerd'); })
        .catch(function(){ toast('⚠️ Import lokaal OK, sync mislukt'); });
    }
    renderClients();
    alert(count+' nieuwe klanten geïmporteerd'+(skip?' ('+skip+' overgeslagen)':'')+'.');
  };
  reader.readAsText(file,'UTF-8');
  e.target.value='';
}

// ── APPT LIST ────────────────────────────────────────────
function renderAppts(){
  var q = (document.getElementById('appt-search')||{}).value || '';
  q = q.toLowerCase();
  var sorted = appts.filter(function(a){
    return !q || clientName(a.clientId).toLowerCase().includes(q);
  }).slice().sort(function(a,b){
    return String(a.date||'').localeCompare(String(b.date||'')) || (a.time||'').localeCompare(b.time||'');
  });
  var el = document.getElementById('appt-list');
  if(!sorted.length){
    el.innerHTML='<div class="empty"><div class="ei">📅</div>Geen afspraken</div>';
    return;
  }
  var mn = ['jan','feb','mrt','apr','mei','jun','jul','aug','sep','okt','nov','dec'];
  el.innerHTML = sorted.map(function(a){
    var dateStr = String(a.date||'').slice(0,10);
    var d = new Date(dateStr+'T12:00');
    var timeDisplay = (a.time && a.time.indexOf('1899') === -1 && a.time.indexOf('GMT') === -1) ? a.time : '';
    var meta = [timeDisplay, a.service, a.price?'€'+a.price:''].filter(Boolean).join(' · ');
    var rowColor = (a.color && colorMap[a.color]) ? colorMap[a.color] : (a.status==='gedaan' ? '#3a3a3a' : '#1D9E75');
    return '<div class="appt-row" style="border-left:4px solid '+rowColor+';padding-left:8px">'+
      '<div class="appt-date-box"><div class="appt-d">'+dateStr.slice(8)+'</div><div class="appt-m">'+(isNaN(d)?'':mn[d.getMonth()])+'</div></div>'+
      '<div class="appt-info"><div class="appt-name">'+clientName(a.clientId)+'</div><div class="appt-meta">'+meta+'</div></div>'+
      '<span class="badge '+(a.status==='gedaan'?'badge-ok':'badge-plan')+'" style="flex-shrink:0">'+a.status+'</span>'+
      '<div style="display:flex;gap:5px;flex-shrink:0">'+
        '<button class="btn-sm" onclick="openApptModal(\''+a.id+'\')">✏️</button>'+
        '<button class="btn-sm btn-danger" onclick="deleteApptById(\''+a.id+'\')">🗑</button>'+
      '</div>'+
    '</div>';
  }).join('');
}

function deleteApptById(id){
  if(!confirm('Afspraak verwijderen?')) return;
  appts = appts.filter(function(x){return x.id!==id;});
  saveLocal();
  api({action:'deleteAppt', id: id}).catch(function(){ toast('⚠️ Niet gesynchroniseerd'); });
  renderAppts(); renderHome(); toast('🗑 Afspraak verwijderd');
}

// ── MODALS ───────────────────────────────────────────────
function closeModals(){
  ['client-modal','appt-wizard','appt-modal'].forEach(function(id){
    document.getElementById(id).classList.add('hidden');
  });
  afterNewClient = false;
}
function bgClose(e, id){ if(e.target===document.getElementById(id)) closeModals(); }

function initColorPicker(pickerId, color){
  var picker = document.getElementById(pickerId);
  if(!picker) return;
  picker.querySelectorAll('.color-opt').forEach(function(opt){
    opt.classList.toggle('selected', opt.dataset.color === (color||''));
    opt.onclick = function(){
      picker.querySelectorAll('.color-opt').forEach(function(o){ o.classList.remove('selected'); });
      opt.classList.add('selected');
    };
  });
}

function getPickerColor(pickerId){
  var picker = document.getElementById(pickerId);
  if(!picker) return '';
  var sel = picker.querySelector('.color-opt.selected');
  return sel ? sel.dataset.color : '';
}

// ── HARD RESET ───────────────────────────────────────────
function hardReset(){
  if(!confirm('Lokale data wissen en alles opnieuw laden van Google Sheets?')) return;
  var activeTab = document.querySelector('.nav button.active');
  var currentTab = activeTab ? activeTab.id.replace('tab-','') : 'home';
  localStorage.removeItem('ct_clients');
  localStorage.removeItem('ct_appts');
  clients = [];
  appts = [];
  api({action:'getAll'}).then(function(data){
    if(data.clients) clients = data.clients;
    if(data.appts)   appts   = normalizeAppts(data.appts);
    saveLocal();
    showTab(currentTab);
    toast('🔄 Gesynchroniseerd');
  }).catch(function(){
    toast('⚠️ Offline — lokale data geladen');
  });
}

// ── INIT ─────────────────────────────────────────────────
// Reload fresh data every time the app becomes visible again
document.addEventListener('visibilitychange', function(){
  if(document.visibilityState === 'visible' && SCRIPT_URL){
    api({action:'getAll'}).then(function(data){
      if(data.clients) clients = data.clients;
      if(data.appts)   appts   = normalizeAppts(data.appts);
      saveLocal();
      // Refresh current visible tab WITHOUT changing tab
      var activeTab = document.querySelector('.nav button.active');
      if(activeTab){
        var tabName = activeTab.id.replace('tab-','');
        if(tabName==='home') renderHome();
        if(tabName==='clients') renderClients();
        if(tabName==='agenda') renderAppts();
      }
    }).catch(function(){ /* offline, keep local */ });
  }
});
